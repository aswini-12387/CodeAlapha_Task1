        <input type="text" id="quiz-answer" placeholder="Enter meaning..." />
        <button class="action" onclick="checkAnswer()">Check</button>
        <p id="quiz-result" class="result"></p>

  <script>
    // Vocabulary List
    const vocabulary = [
      { word: "Bonjour", meaning: "Hello" },
      { word: "Merci", meaning: "Thank you" },
      { word: "Chien", meaning: "Dog" },
      { word: "Chat", meaning: "Cat" },
      { word: "Pomme", meaning: "Apple" },
      { word: "Pain", meaning: "Bread" }
    ];

    // Grammar Tips
    const grammarTips = [
      "In French, adjectives usually follow the noun (e.g., 'une pomme rouge' = a red apple).",
      "Use 'le' for masculine nouns and 'la' for feminine nouns.",
      "'√ätre' means 'to be' and is an irregular verb.",
      "'Avoir' means 'to have' and is also irregular.",
      "To form questions, invert the subject and verb (e.g., 'Parlez-vous fran√ßais?')."
    ];

    // Track current index
    let vocabIndex = 0;
    let grammarIndex = 0;

    function switchTab(tabId) {
      document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
      document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
      document.getElementById(tabId).classList.add('active');
      event.target.classList.add('active');
      if (tabId === 'quiz') loadQuiz();
    }

    function nextVocab() {
      vocabIndex = (vocabIndex + 1) % vocabulary.length;
      updateVocab();
    }

    function updateVocab() {
      const item = vocabulary[vocabIndex];
      document.getElementById("vocab-word").textContent = item.word;
      document.getElementById("vocab-meaning").textContent = item.meaning;
    }

    function speakWord() {
      const word = vocabulary[vocabIndex].word;
      const utterance = new SpeechSynthesisUtterance(word);
      utterance.lang = "fr-FR";
      speechSynthesis.speak(utterance);
    }

    function nextGrammar() {
      grammarIndex = (grammarIndex + 1) % grammarTips.length;
      document.getElementById("grammar-tip").textContent = grammarTips[grammarIndex];
    }

    function loadQuiz() {
      const random = vocabulary[Math.floor(Math.random() * vocabulary.length)];
      document.getElementById("quiz-word").textContent = random.word;
      document.getElementById("quiz-word").dataset.answer = random.meaning.toLowerCase();
      document.getElementById("quiz-answer").value = '';
      document.getElementById("quiz-result").textContent = '';
    }

    function checkAnswer() {
      const answer = document.getElementById("quiz-answer").value.trim().toLowerCase();
      const correct = document.getElementById("quiz-word").dataset.answer;
      const result = document.getElementById("quiz-result");
      if (answer === correct) {
        result.textContent = "‚úÖ Correct!";
        result.style.color = "green";
      } else {
        result.textContent = `‚ùå Incorrect. Correct: ${correct}`;
        result.style.color = "red";
      }
      setTimeout(loadQuiz, 2000);
    }

    // Initial loads
    updateVocab();
    nextGrammar();
  </script>
